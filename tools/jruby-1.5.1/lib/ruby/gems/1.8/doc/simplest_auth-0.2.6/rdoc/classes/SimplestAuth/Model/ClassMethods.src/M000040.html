      def authenticate_by(ident)
        if active_record?
          instance_eval "def authenticate(\#{ident}, password)\nklass = find_by_\#{ident}(\#{ident})\n(klass && klass.authentic?(password)) ? klass : nil\nend\n"
        elsif data_mapper? || mongo_mapper?
          instance_eval "def authenticate(\#{ident}, password)\nklass = first(:\#{ident} => \#{ident})\n(klass && klass.authentic?(password)) ? klass : nil\nend\n"
        end
      end
